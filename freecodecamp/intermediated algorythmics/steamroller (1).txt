
function steamrollArray(arr) {
  let flattedArray = [];
 let flat = function (arg) {
   if(!Array.isArray(arg)) {
     flattedArray.push(arg);
     console.log(flattedArray);
   }
   else {
     arg.forEach(flat);
   }
 }
 arr.forEach(flat);
  return flattedArray;
}
 


steamrollArray([1, {}, [3, [[4]]]])
